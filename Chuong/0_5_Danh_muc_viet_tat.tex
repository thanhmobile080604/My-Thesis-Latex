\begin{longtable}{p{2.5cm} p{5.3cm} p{6.2cm}}
	\hline
   \textbf{Viết tắt}  & \textbf{Tên tiếng Anh}	& \textbf{Tên tiếng Việt} \\ \hline
	\textbf{API} & Application Programming Interface & Giao diện lập trình ứng dụng\\
	\textbf{AR} & Augmented Reality & Thực tế tăng cường\\
	\textbf{CPU} & Central Processing Unit & Bộ xử lý trung tâm\\
	\textbf{ECEF} & Earth-Centered, Earth-Fixed & Hệ tọa độ gắn với Trái Đất, gốc tại tâm Trái Đất\\
	\textbf{ENU} & East-North-Up & Hệ tọa độ cục bộ Đông-Bắc-Lên\\
	\textbf{FBE} & Forward-Backward Error & Sai số tiến-lùi trong kiểm tra Optical Flow\\
	\textbf{FPS} & Frames Per Second & Số khung hình xử lý trong một giây\\
	\textbf{GNSS} & Global Navigation Satellite System & Hệ thống vệ tinh dẫn đường toàn cầu\\
	\textbf{GPS} & Global Positioning System & Hệ thống định vị toàn cầu của Hoa Kỳ\\
	\textbf{GPU} & Graphics Processing Unit & Bộ xử lý đồ họa\\
	\textbf{IGS} & International GNSS Service & Dịch vụ GNSS quốc tế\\
	\textbf{IMU} & Inertial Measurement Unit & Cụm cảm biến đo lường quán tính\\
	\textbf{KLT} & Kanade-Lucas-Tomasi & Bộ theo dõi đặc trưng dựa trên Lucas-Kanade và Shi-Tomasi\\
	\textbf{LLA} & Latitude, Longitude, Altitude & Vĩ độ, kinh độ, độ cao\\
	\textbf{ONNX} & Open Neural Network Exchange & Định dạng trao đổi mô hình mạng neural\\
	\textbf{PVT} & Position, Velocity, Time & Vị trí, vận tốc và thời gian\\
	\textbf{RAFT} & Recurrent All-Pairs Field Transforms & Mô hình mạng neural ước lượng Optical Flow\\
	\textbf{ROI} & Region of Interest & Vùng quan tâm trong ảnh\\
	\textbf{SGP4} & Simplified General Perturbations 4 & Mô hình lan truyền quỹ đạo từ phần tử trung bình\\
	\textbf{TAI} & International Atomic Time & Thời gian nguyên tử quốc tế\\
	\textbf{TLE} & Two-Line Element & Bộ phần tử quỹ đạo hai dòng\\
	\textbf{UTC} & Coordinated Universal Time & Giờ phối hợp quốc tế\\
	\textbf{WGS84} & World Geodetic System 1984 & Hệ quy chiếu trắc địa thế giới 1984\\
    \hline
\end{longtable}
